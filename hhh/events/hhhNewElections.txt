namespace = elections_hhh

country_event = {
	id = elections_hhh.1
	title = elections_hhh.1.t
	desc = elections_hhh.1.d
	picture = LIBERUM_VETO_eventPicture
	
	trigger = {
	    has_regency = no
		has_states_general_mechanic = yes
		has_reform = greek_monarchy
	}
	
	is_triggered_only = yes
	
	immediate = {}
	
	option = {
		name = hhh.elections.a
		 trigger = {
			legitimacy = 5
			has_regency = no
		}
		custom_tooltip = remains_ruler
		if = {
			limit = {
			is_statists_in_power = yes
			}
			add_ruler_modifier = {
					name = "Archon_resists"
					duration = 1595
				}
			add_estate_influence_modifier = {
					estate = estate_hhh
					desc = EST_VAL_HHH_ELECT
					influence = -15
					duration = 5440
				}
			add_estate_loyalty = {
					estate = estate_hhh
					loyalty = -20
				}
			
			add_legitimacy = -10	
			change_statists_vs_orangists = 0.07	
        }
        if = {
			limit = {
				is_orangists_in_power = yes
			}
			add_estate_influence_modifier = {
					estate = estate_hhh
					desc = EST_VAL_HHH_ELECT
					influence = -10
					duration = 5440
			}
			add_estate_loyalty = {
				estate = estate_hhh
				loyalty = -10
			}	
        }	
		add_legitimacy = -10
		change_statists_vs_orangists = 0.33
	}	
	option = {
		name = hhh.elections.b
		define_ruler = {
			dynasty = ROOT
		}
		add_estate_influence_modifier = {
				estate = estate_hhh
				desc = EST_VAL_HHH_ELECT
				influence = 15
				duration = 3100
		}
		add_ruler_modifier = {
				name = "Archon_take_power"
				duration = 365
		}	
		add_estate_loyalty = {
				estate = estate_hhh
				loyalty = 20
		}
		change_statists_vs_orangists = -0.33
	}
	option = {
		name = hhh.elections.d
		define_ruler = {
			dynasty = ROOT
		}
		add_estate_influence_modifier = {
				estate = estate_hhh
				desc = EST_VAL_HHH_ELECT
				influence = -10
				duration = 3100
		}
		add_estate_loyalty = {
				estate = estate_hhh
				loyalty = -20
		}
		change_statists_vs_orangists = 0.33
	}
}

country_event = {
	id = elections_hhh.3
	title = elections_hhh.3.t
	desc = elections_hhh.3.d
	picture = LIBERUM_VETO_eventPicture
	
	trigger = {
	    has_regency = no
		has_states_general_mechanic = yes
		has_reform = greek_monarchy
		NOT = { 
			has_ruler_modifier  = archon_elect
			has_country_flag = event_council_flag
			}
	}
	
	immediate = {
		add_ruler_modifier = { name = "archon_elect" duration = 5475 }
		set_country_flag = event_council_flag
	}
	
		
	option = {
		name = hhh.elections3.a
		change_statists_vs_orangists = -0.15
		hidden_effect = {
		        clr_country_flag = event_council_flag
		}
	}
}
